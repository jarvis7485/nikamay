class Addition {
    int add(int a, int b) {
        return a + b;
    }
    
    float add(float a, float b, float c) {
        return a + b + c;
    }
    
    String add(String a, String b) {
        return a + b;
    }
    
    public static void main(String[] args) {
        Addition addition = new Addition();
        System.out.println("Addition of 2 integers: " + addition.add(3, 4));
        System.out.println("Addition of 3 floats: " + addition.add(1.1f, 2.2f, 3.3f));
        System.out.println("Addition of 2 strings: " + addition.add("Hello", "World"));
    }
}
